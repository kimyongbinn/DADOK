create or replace FUNCTION order_nb_title
   (p_O_ORDER_NUM    in number)
    RETURN  VARCHAR2  -- 상품제목(신/중고)

IS   
    v_nb_num        number;                 -- 상품번호(신/중고)
    v_title         NEWBOOK.nb_title%TYPE;  -- 상품제목(신/중고)
    v_O_DE_PRODTYPE number;                 -- 상품유형

BEGIN
   SELECT MIN(nb_num)
   INTO   v_nb_num 
   FROM   order_detail
   WHERE  O_ORDER_NUM = p_O_ORDER_NUM
  ;

   --  v_nb_num < 200000 이면 신상품, 아니면 중고상품
   IF      v_nb_num < 200000
   THEN
           SELECT nb_title
           INTO   v_title
           FROM   newbook
           WHERE  nb_num=v_nb_num;
   ELSE      
           SELECT nb_title
           INTO   v_title
           FROM   Newbook
           WHERE  nb_num = (
                     SELECT nb_num
                     FROM   oldbook
                     WHERE  ob_num=v_nb_num);   
    END IF;

    RETURN v_title;
END;

create or replace FUNCTION order_nb_price
   (p_O_ORDER_NUM    in number)
    RETURN  NUMBER  -- 상품금액 (상품 가격 합산)

IS   
    v_nb_num        NUMBER;     -- 상품번호(신/중고)
    v_nb_price      NUMBER;     -- 상품가격(신/중고)
    v_nb_price_sum  NUMBER;     -- 상품금액(상품 가격 합산)

    CURSOR  order_detail_cursor  IS
    SELECT   nb_num, O_DE_COUNT, O_DE_PRODTYPE
    FROM     order_detail
    WHERE    O_ORDER_NUM = p_O_ORDER_NUM
     ;
BEGIN
    DBMS_OUTPUT.ENABLE;
    -- 초기값 설정
    v_nb_price  := 0; 
    v_nb_price_sum  := 0; 
    
    -- row 별 반복 수행
    FOR   order_detail IN order_detail_cursor LOOP
       --  order_detail.O_DE_PRODTYPE = 1 이면 신상품, 아니면 중고상품
       IF      order_detail.O_DE_PRODTYPE = 1
       THEN
               SELECT nb_price
               INTO   v_nb_price
               FROM   newbook
               WHERE  nb_num=order_detail.nb_num;
       ELSE      
               SELECT ob_pur_price * 1.1
               INTO   v_nb_price
               FROM   Oldbook
               WHERE  ob_num = order_detail.nb_num;   
        END IF;
        v_nb_price := (order_detail.O_DE_COUNT * v_nb_price);
        v_nb_price_sum := v_nb_price_sum + v_nb_price;
    END LOOP;

    RETURN v_nb_price_sum;
 
    EXCEPTION
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE(SQLERRM||'에러 발생 ');
END;

create or replace FUNCTION max_orderr_num
   RETURN number 
   -- 함수값은 return 값 있어야함, out 값 X, return값 2개도 못주고 무조건 한개
   -- 독립성 보장 (FUNCTION을 DML로 쓰고싶을 때) : 암기할 필요없고 가지고다니면서 쓰라함
IS  
    vcurDate number(20);
    vCount number(3) := 0;
BEGIN
   Select TO_NUMBER(TO_CHAR(SYSDATE, 'YYYYMMDD'))
   INTO   vcurDate
   From   dual;

   SELECT COUNT(*)
   INTO   vCount
   FROM   orderr 
   WHERE  substr (O_ORDER_num,1,8) = vcurDate
   ;

   IF        vCount > 0      THEN
             SELECT MAX(O_ORDER_num) + 1
             INTO   vcurDate
             FROM   orderr;
  ELSE      
             SELECT  TO_NUMBER(TO_CHAR(SYSDATE, 'YYYYMMDD') || '0001')
             INTO   vcurDate
             FROM   dual;
   END IF; 

    RETURN vcurDate;
END max_orderr_num;

----관리자 메인페이지 월별 주문 건수 프로시저 create문------

create or replace Procedure  cur_mon_order_cnt
( mon_order_cnt_cursor    OUT     SYS_REFCURSOR )
IS
    v_sdate  DATE;      -- 올해 시작일
    v_edate  DATE;      -- 올해 종료일
    v_syymm  VARCHAR2(7);  -- 올해 시작연월
    v_eyymm  VARCHAR2(7);  -- 올해 종료연월

Begin
    SELECT TO_DATE(to_char(sysdate,'yyyy')||'-01-01', 'YYYY-MM-DD'), TO_DATE(to_char(sysdate,'yyyy')||'-12-31', 'YYYY-MM-DD'), to_char(sysdate,'yyyy')||'-01' , to_char(sysdate,'yyyy')||'-12'
    INTO   v_sdate                       , v_edate                         , v_syymm                      , v_eyymm
    from dual;


 -- 1.DECLARE/OPEN/FETCH 단계
  OPEN mon_order_cnt_cursor 
  FOR
    SELECT 
        TO_CHAR(NVL(a.cnt1, 0)) cnt1
    FROM (
          SELECT TO_CHAR(o_order_date, 'yyyy-mm') as MONTHS1, COUNT(*) as cnt1
          FROM orderr
          WHERE o_order_date BETWEEN v_sdate AND v_edate
          GROUP BY TO_CHAR(o_order_date, 'yyyy-mm')
          ) a,
          (
          SELECT TO_CHAR(ADD_MONTHS(v_sdate, LEVEL - 1 ),'YYYY-MM') AS MONTHS1
          FROM dual
          CONNECT BY LEVEL <= MONTHS_BETWEEN( TO_DATE(v_eyymm, 'YYYY-MM'), TO_DATE(v_syymm, 'YYYY-MM') ) + 1
          ) b
    WHERE b.MONTHS1 = a.MONTHS1(+)
    ORDER BY b.MONTHS1;


End cur_mon_order_cnt;
